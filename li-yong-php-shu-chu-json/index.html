if($result){

    {
        $user = new User();
        $user->id = $row["id"];
        $user->username = $row["username"];
        $user->password = $row["password"];
        $data[]=$user;
    }
//    $json = json_encode($data);因为这条语句无法输出中文，所以换成下面这个语句
    $json = json_encode($data,JSON_UNESCAPED_UNICODE);
    echo "{".'"user"'. ":".$json." }";
}else{
    echo "查询失败";
}

六，整体代码及效果

<?php
include_once ("conn.php");
$json = '';
$data = array();
class  User{
    public $id;
    public $username;
    public $password;
}
$sql="select * from wlb";
$result = $con->query($sql);
if($result){
    while ($row = mysqli_fetch_array($result,MYSQLI_ASSOC))
    {
        $user = new User();
        $user->id = $row["id"];
        $user->username = $row["username"];
        $user->password = $row["password"];
        $data[]=$user;
    }
//    $json = json_encode($data);
    $json = json_encode($data,JSON_UNESCAPED_UNICODE);
    echo "{".'"user"'. ":".$json." }";
}else{
    echo "查询失败";
}
?>
